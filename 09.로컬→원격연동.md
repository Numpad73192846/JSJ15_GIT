# 로컬 저장소 생성 후 원격 저장소와 연결하기

0. git config --global user.name "사용자 이름"
   git config --global user.email "사용자 이메일"
1. git init
2. git remote add origin <URL>
3. git fetch
4. 새 파일 생성
5. git add .
6. git commit -m "메시지"
7. git branch --set-upstream-to=origin/main main
8. git merge --allow-unrelated-histories origin/main
9. git push
10. 원격 저장소에서 새 파일 생성
11. git pull
